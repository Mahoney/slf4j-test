14  class OverridableProperties {

16      private final String propertySourceName;

18  
19      OverridableProperties(final String propertySourceName) throws IOException {
20          this.propertySourceName = propertySourceName;
21          this.properties = getProperties();
22      }
23  
24      private Properties getProperties() throws IOException {
25          final Optional<InputStream> resourceAsStream = fromNullable(Thread.currentThread().getContextClassLoader()
26                  .getResourceAsStream(propertySourceName + ".properties"));
27          return resourceAsStream.transform(new Function<InputStream, Properties>() {
28              @Override
29              public Properties apply(final InputStream propertyResource) {
30                  Properties loadedPropertoes = new Properties();
31                  try (InputStream closablePropertyResource = propertyResource) {
32                      loadedPropertoes.load(closablePropertyResource);
33                  } catch (IOException ioe) {
34                      Exceptions.throwUnchecked(ioe);
35                  }
36                  return loadedPropertoes;
37              }
38          }).or(EMPTY_PROPERTIES);
39      }
40  
41      String getProperty(final String propertyKey, final String defaultValue) {
42          final String propertyFileProperty = properties.getProperty(propertyKey, defaultValue);
43          return System.getProperty(propertySourceName + "." + propertyKey, propertyFileProperty);
44      }
