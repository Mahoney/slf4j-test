6   import org.slf4j.spi.MDCAdapter;

10  import uk.org.lidalia.lang.ThreadLocal;

12  import static com.google.common.base.Optional.fromNullable;
13  
14  public class TestMDCAdapter implements MDCAdapter {
15  
16      private final ThreadLocal<Map<String, String>> value = new ThreadLocal<>(
17              Suppliers.<String, String>makeEmptyMutableMap());
18  
19      public void put(final String key, final String val) {
20          value.get().put(key, fromNullable(val).or("null"));
21      }
22  
23      public String get(final String key) {
24          return value.get().get(key);
25      }
26  
27      public void remove(final String key) {
28          value.get().remove(key);
29      }
30  
31      public void clear() {
32          value.get().clear();
33      }
34  
35      public ImmutableMap<String, String> getCopyOfContextMap() {
36          return ImmutableMap.copyOf(value.get());
37      }
38  
39      @SuppressWarnings("unchecked")
40      public void setContextMap(final Map contextMap) {
41          value.set(new HashMap<String, String>(contextMap));
42      }
43  }
