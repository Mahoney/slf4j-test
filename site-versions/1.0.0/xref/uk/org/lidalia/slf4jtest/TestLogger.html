31  /**
32   * Implementation of {@link Logger} which stores {@link LoggingEvent}s in memory and provides methods
33   * to access and remove them in order to facilitate writing tests that assert particular logging calls were made.
34   * <p/>
35   * {@link LoggingEvent}s are stored in both an {@link ThreadLocal} and a normal {@link List}. The {@link #getLoggingEvents()}
36   * and {@link #clear()} methods reference the {@link ThreadLocal} events. The {@link #getAllLoggingEvents()} and
37   * {@link #clearAll()} methods reference all events logged on this Logger.  This is in order to facilitate parallelising
38   * tests - tests that use the thread local methods can be parallelised.
39   * <p/>
40   * By default all Levels are enabled.  It is important to note that the conventional hierarchical notion of Levels, where
41   * info being enabled implies warn and error being enabled, is not a requirement of the SLF4J API, so the
42   * {@link #setEnabledLevels(ImmutableSet)}, {@link #setEnabledLevels(Level...)},
43   * {@link #setEnabledLevelsForAllThreads(ImmutableSet)}, {@link #setEnabledLevelsForAllThreads(Level...)} and the various
44   * isXxxxxEnabled() methods make no assumptions about this hierarchy.  If you wish to use traditional hierarchical setups you may
45   * do so by passing the constants in {@link uk.org.lidalia.slf4jext.ConventionalLevelHierarchy} to
46   * {@link #setEnabledLevels(ImmutableSet)} or {@link #setEnabledLevelsForAllThreads(ImmutableSet)}.
47   */

433     private void addLoggingEvent(
434             final Level level,
435             final Optional<Marker> marker,
436             final Optional<Throwable> throwable,
437             final String format,
438             final Object... args) {
439         if (enabledLevels.get().contains(level)) {
440             final LoggingEvent event = new LoggingEvent(of(this), level, mdc(), marker, throwable, format, args);
441             allLoggingEvents.add(event);
442             loggingEvents.get().add(event);
443             testLoggerFactory.addLoggingEvent(event);
444             optionallyPrint(event);
445         }
446     }

448     @SuppressWarnings("unchecked")
449     private Map<String, String> mdc() {
450         return fromNullable(MDC.getCopyOfContextMap()).or(Collections.emptyMap());
451     }

453     private void optionallyPrint(final LoggingEvent event) {
454         if (testLoggerFactory.getPrintLevel().compareTo(event.getLevel()) <= 0) {
455             event.print();
456         }
457     }
